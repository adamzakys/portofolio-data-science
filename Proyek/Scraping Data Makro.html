with open(file_path, 'r') as f:
    html_content = f.read()

soup = BeautifulSoup(html_content, 'html.parser')

rows = soup.find_all('tr', class_='js-event-item')

data = []
for row in rows:
    date = row['data-event-datetime'].split()[0]
    currency = row.find('td', class_='flagCur').text.strip()
    event = row.find('td', class_='event').text.strip()
    actual_element = row.find('td', class_=lambda c: c and 'actual' in c)
    if actual_element:
        actual = actual_element.text.strip()
    else:
        actual = "Not Found"

    data.append([date, currency, event, actual])  # Append only desired columns

with open('data_ekstrak.csv', 'w', newline='', encoding='utf-8') as csvfile:
    writer = csv.writer(csvfile)
    writer.writerow(['Tanggal', 'Mata Uang', 'Event', 'Aktual'])  # Updated header
    writer.writerows(data)

print("Data telah disimpan ke file 'data_ekstrak.csv'")
